# 防抖和节流

## 防抖

当事件触发时，设定一个时间延迟执行，若器间又触发，则重新设定时间。

```html
// 比如连续的点击发送

<button>发送</button>

<script>
	let btn = document.querySelector('button');
  let do = debounce(callback, 300);
  btm.onclick = do;
  
  function callback(){
  	console.log(this);
  }
  
  function debounce(func, time) {
  	let timer;
  	return function() {
    	clearTimeout(timer);
     	timer = setTimeOut(() => {func.call(this)}, time);
    }
  }
</script>
```

## 节流

规定在一个周期内，只能触发一次函数。	如果多次触发，只有一次生效。

```html
<button>发送</button>

<script>
	let btn = document.querySelector('button');
  btm.onclick = function() {
  	throttle(callback, 300);
  };
  
  function callback(){
  	console.log(this);
  }
  
  function throttle(func, time) {
  	let timeOut;
    let startTime = new Date();
  	return function() {
      let curTime = new Date();
      // 清除定时器
      clearTimeOut(timeOut);
      // 如果已经过了指定时间，调用时间处理函数
    	if (curTime - startTime >= time) {
    		func.call(this);
        startTime = new Date();
      }
      // 如果没有，重新计算剩余时间
      else {
      	timeOut = setTimeOut(() => {
        	func.call(this);
          startTime = new Date();
        }, curTime - startTime);
      }
    }
  }
</script>
```

## 使用场景

- 防抖：高频率的点击发送；
- 节流：代码周期性执行，如resize事件；
