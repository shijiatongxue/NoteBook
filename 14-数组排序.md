# sort()
```js
// 从小到大排序
var arr = [1, 2, 3, 4, 5, -1, 10, 15, 8];
var sortArray = function(a, b){
    return a - b;
}
arr = arr.sort(sortArray);
```

# 冒泡排序
```js
// 同样是从小到大排序
Array.prototype.bubbleSort = function(){
    // 外循环控制循环次数
    for(let i = 0, len = this.length; i < len; i++){
        for (let j = 0; j < len-i-1; j++){
            if(this[j+1] < this[j]){
                let tmp = this[j+1];
                this[j+1] = this[j];
                this[j] = tmp;
            }
        }
    }
}
var arr = [1,2,3,4,5,4,3,2,1,-1,100];
arr.bubbleSort();
```
# 快速排序
```js
// 不定义到原型里
var quickSort = function(arr){
    if(arr.length < 2){
        return arr;
    }
    let mid = Math.floor(arr.length/2);
    let left = [];
    let right = []; 
    for(let i = 0, len = arr.length; i < len; i++){
        if(arr[i] < arr[mid]){
            left.push(arr[i]);
        }
        else if(arr[i] > arr[mid]){
            right.push(arr[i]);
        }
    }
    return quickSort(left).concat(arr[mid], quickSort(right));
}
```
