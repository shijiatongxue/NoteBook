# 拍平数组

```js
function flatten(arr){
    let res = [];
    fc(arr);
    
    function fc(arr){
        for (item of arr) {
            if (!Array.isArray(item)) {
                res.push(item);
            } else {
              for (innerItem of item) {
                  if (!Array.isArray(innerItem)){
                      res.push(innerItem);
                  } else {
                      fc(innerItem);
                  }
              }
            }
        }
    }
    return res;
}

flatten([1, {}, [2, 3, [4]]]);
```
