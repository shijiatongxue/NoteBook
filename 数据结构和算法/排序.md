## 冒泡排序

```js
// 冒泡排序

function bubbleSort(arr) {
  if (arr === null || arr.length === 0) {return arr;}
  
  for (let i = 0, len = arr.length; i < len-1; i++) {
    let flag = false;
  	for (let j = 0; j < len-i-1; j++) {
    	if (arr[j] > arr[j+1]) {
      	[arr[j], arr[j+1]] = [arr[j+1], arr[j]];
        flag = true;
      }
    }
   	if (flag === false) {break;}
  }
  return arr;
}
```

平均时间复杂度：O(N^2)

最优时间复杂度：O(N)

时间复杂度：O(1)

## 选择排序

```js
function selectSort(arr) {
  if (arr === null || arr.length === 0) {return arr;}
  
  for (let i = 0, len = arr.length; i < len-1; i++) {
  	for (let j = i+1; j < len; j++) {
    	if (arr[i] > arr[j]) {
    		[arr[j], arr[i]] = [arr[i], arr[j]];
      }
    }
  }
  return arr;
}
```

时间复杂度：O(N^2)

空间复杂度：O(1)

## 插入排序

```js
function insertSort(arr) {
  if (arr === null || arr.length === 0) {return arr;}
  
  for (let i = 1, len = arr.length; i < len; i++) {
  	for (let j = i-1; j >= 0; j--) {
    	if (arr[j] > arr[j+1]) {
      	[arr[j], arr[j+1]] = [arr[j+1], arr[j]];
      } else {
        break;
      }
    }
  }
  return arr;
}
```

最优时间复杂度：O(N)

平均时间复杂度：O(N^2)

空间复杂度：O(1)

## 快速排序

```js
function quickSort(arr) {
	if (arr.length <= 1)   {
  	return arr;
  }
  
  let now = arr[0];
  let left = [];
  let right  = [];
  for (let i = 1, len = arr.length; i < len; i++) {
  	if (arr[i] < now) {
   		left.push(arr[i]);
   	} else {
    	right.push(arr[i]);
    }
  }
  
  return quickSort(left).concat(now).concat(quickSort(right));
}
```

平均时间复杂度：O(NlogN)

最坏时间复杂度：O(N^2)

## 归并排序

```js
function mergeSort(arr){
 	let len = arr.length;
  if (len <= 1) {
  	return arr;
  }
  if (len === 2) {return merge([arr[0]], [arr[1]]);}
  return merge(mergeSort(arr.slice(0, +(len/2))), mergeSort(arr.slice(+(len/2))));
  
  // 两个有序数组的合并
  function merge(a, b) {
  	let len1 = a.length;
    let len2 = b.length;
    let res = [];
   	let i = 0, j = 0;
    while (i < len1 && j < len2) {
    	if (a[i] < b[j]) {
     		res.push(a[i]);
        i++;
      } else {
      	res.push(b[j]);
        j++;
      }
    }
    if (i >= len1) {
    	res = res.concat(b.slice(j));
    } else if (j >= len2){
      res = res.concat(a.slice(i));
    }
    return res;
  }
}
```

## 堆排

```js
function heapSort(arr) {
  // 建堆
  for (let i = Math.floor(arr.length/2-1); i >= 0; i--) {
 		shiftDown(i);
  }
  
  // 依次拿出最大元素
  for (let i = arr.length-1; i > 0; i--) {
  	[arr[0], arr[i]] = [arr[i], arr[0]];
   	shiftDown(0, i); // 调整堆
  }
  
  // 把i节点以下的堆整理为最大堆
  function shiftDown(i, len=arr.length) {
    for (let j = 2*i+1; j < len;) {
      if (j + 1 < len && arr[j] < arr[j+1]) { // 这里要判断j是否出界
    		j++; // 找到较大的一个
      }
      if (arr[i] < arr[j]) {
      	[arr[i], arr[j]] = [arr[j], arr[i]];
        i = j;
      } else {
      	break;
      }
    }
  }
  
  return arr;
}
```

时间复杂度：O(NlogN)

空间复杂度：O(N)
